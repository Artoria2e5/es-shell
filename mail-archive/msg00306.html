Hi rog,
	Don't know about compiling es on ultrix, but here's my quick hacks
to use fifos rather than tmp files.

Pete.

------- 8< ------------------------- 8< ------------------------

fn %readfrom var cmd body {
	local ($var = /tmp/es.$var.$pid) {
		unwind-protect {
			mknod $$var p
			$&background {$cmd > $$var; exit}
			$body
		} {
			rm -f $$var
		}
	}
}

fn %writeto var cmd body {
	local ($var = /tmp/es.$var.$pid) {
		unwind-protect {
			mknod $$var p
			$&background {$cmd < $$var; exit}
			$body
		} {
			rm -f $$var
		}
	}
}
